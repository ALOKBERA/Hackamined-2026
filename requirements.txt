# SnapSense AI — Requirements
# ============================================================
# MERN Stack: MongoDB + Express + React + Node.js
# ============================================================


# ─── SYSTEM REQUIREMENTS ─────────────────────────────────────
Node.js        >= 18.x         https://nodejs.org
npm            >= 9.x          (comes with Node.js)
MongoDB        >= 7.x          https://www.mongodb.com/try/download/community
Git            any             https://git-scm.com


# ─── API KEYS REQUIRED ───────────────────────────────────────
# All go into server/.env

1. GROQ API KEY
   Get it from: https://console.groq.com
   Used for: AI image classification (llama-4-scout vision model)

2. GOOGLE OAUTH CREDENTIALS (Client ID + Secret)
   Get it from: https://console.cloud.google.com
   Steps:
     - Create project → Enable APIs:
         * Google Drive API
         * Google Sheets API
         * Google Calendar API
     - OAuth 2.0 consent screen → External → add test users
     - Create Credentials → OAuth 2.0 Client ID → Web Application
     - Authorized redirect URI: http://localhost:5000/auth/google/callback

3. MONGODB URI
   Local:  mongodb://localhost:27017/snapsense   (requires MongoDB installed)
   Atlas:  mongodb+srv://<user>:<pass>@cluster.mongodb.net/snapsense
           Get it from: https://cloud.mongodb.com (free tier available)

# ─── ENVIRONMENT VARIABLES (server/.env) ─────────────────────

NODE_ENV=development

GOOGLE_CLIENT_SECRET=<your_google_client_secret>

# ─── HOW TO INSTALL ALL NPM PACKAGES ─────────────────────────
cd server && npm install
cd ../client && npm install


# ─── HOW TO RUN ──────────────────────────────────────────────
# Terminal 1 (start MongoDB if not running as service):
Start-Process "C:\Program Files\MongoDB\Server\8.2\bin\mongod.exe" -ArgumentList "--dbpath C:\data\db"

# Terminal 2 (backend):
cd server && npm run dev      # runs on http://localhost:5000

# Terminal 3 (frontend):
cd client && npm run dev      # runs on http://localhost:5173
